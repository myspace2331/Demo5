To apply navigation and routing in flutter app
(To develop a flutter app with two screens, screen-1 should navigate to
screen-2.)

import 'package:flutter/material.dart';

void main() {
  runApp(MaterialApp(
    home: HomeRoute(),
  ));
}

class HomeRoute extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      // Background for the first page
      backgroundColor: Colors.blueGrey[50],
      appBar: AppBar(
        title: Text('APPLICATION NAME'),
        backgroundColor: Colors.green,
      ),
      body: Center(
        child: ElevatedButton(
          child: Text('Click Me!'),
          onPressed: () {
            Navigator.push(
              context,
              MaterialPageRoute(builder: (context) => SecondRoute()),
            );
          },
        ),
      ),
    );
  }
}

class SecondRoute extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      // Background for the second page
      backgroundColor: Color(0xffffb2fb),
      appBar: AppBar(
        title: Text("HOME CLICK"),
        backgroundColor: Color(0x8698e3ff),
      ),
      body: Center(
        child: ElevatedButton(
          onPressed: () {
            Navigator.pop(context);
          },
          child: Text('Home!'),
        ),
      ),
    );
  }
}
